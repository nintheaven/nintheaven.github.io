mysql用户与权限管理

: 未分类
发布于 : 2022-10-11

用户管理

mysql用户相关信息存储在名为mysql的库中的user表中

查看所有用户

use mysql;
select user from user;

查看当前用户

select user();

创建用户账号

create user <用户名> identified by '<密码>';

修改用户名

rename user <用户名> to <新用户名名>;

修改密码

alter user <用户名> identified by '<新密码>';

删除用户账号

drop user <用户名>;

权限管理

查看用户权限

show grants;
show grants for <用户名>;

赋予权限

grant 权限 on 数据库和表 to 用户名;

grant all on db0.* to lyp;

撤销权限

revoke all on db0.* from lyp;
