element-plus

安装

npm install element-plus --save

引入

在main.js中

import { createApp } from 'vue'
import App from './App.vue'

import ElementPlus from 'element-plus';

文件上传

手动上传

<template>
    <el-upload action="/admin/image" ref="upload" name="file" list-type="picture-card"
        :auto-upload="false"
        :before-upload="beforeUpload"
        :http-request="onUpload">
        <template #trigger>
            <el-button size="small" type="text">选取图片</el-button>
        </template>
        <template #tip>
            <div class="el-upload__tip">
                <small>只能上传jpg/png文件,且不超过1MB</small>
            </div>
        </template>
        <div>
            <el-button style="margin: 10px 0 0 10px;" size="small" type="success" @click="submitUpload">上传到服务器</el-button>
        </div>
    </el-upload>
<template>

<script>
import http from '@/static/js/http'
import {successMsg,errorMsg} from '@/static/js/ui'

import Title from '@/components/Title.vue'

export default {
    name: 'Image',

    components:{
        
    },

    data() {
        return {
            
        };
    },

    methods: {
        // 上传之前的相关处理逻辑
        beforeUpload(file) {
            const isImage = (file.type === 'image/jpeg') || (file.type === 'image/png');
            const isLt2M = file.size / 1024 / 1024 <= 1;

            if (!isImage) {
                errorMsg('上传头像图片只能是JPG/PNG格式!');
            }
            if (!isLt2M) {
                errorMsg('上传头像图片大小不能超过1MB!');
            }
            return isImage && isLt2M;
        },

        // 上传图片
        onUpload(input){
            let url=input.action;
            let body=new FormData();
            body.append('file',input.file);
            let config={
                'Content-Type': 'multipart/form-data'
            };
            http.post(url,body,config).then(response=>{
                let result=response.data;
                if (result.ok) {
                    successMsg(result.message);
                }
            });
        },

        submitUpload() {
            this.$refs.upload.submit();
        },
    }
}
</script>

el-switch开关的用法

现在要实现的功能是，在数据列表中，每条数据都需要有至少一个开关，switch状态发生变化时的回调函数@change,除了开关状态外，还要当前行数据的信息，例如id。

@change回调函数默认只有表示开关状态的一个参数，要想给回调函数添加其它参数，需要使用$event获取开关状态，再添加其它参数即可。

实现方式如下

<template>
    <el-table :data="page.items" style="width: 100%" height="80vh" size="medium" fit stripe>
        <el-table-column label="公开">
            <template #default="scope">
                <el-switch v-model="scope.row.enabled" @change="onEnabledChanged($event,scope.row.id)"
                    active-color="#13ce66" inactive-color="#dcdfe6" />
            </template>
        </el-table-column>
    </el-table>
</template>

<script>
export default {
    methods: {
        // 设置是否公开
        onEnabledChanged(enabled,articleId){
            // ...
        },
    }
}
</script>

el-form报错问题

只要在表单中使用任何form-item，都会疯狂报错，代码如下

<el-form :model="addTagForm" size="mini">
    <el-form-item label="名称">
        <el-input v-model="addTagForm.name"></el-input>
    </el-form-item>
</el-form>

报错

index.esm.js?3fd4:1 ElementPlusError: [Util] binding value must be a string or number

解决

MLGB,是element-plus版本 1.0.2-beta.55 中的bug,回退版本

npm uninstall element-plus

npm install -s element-plus@^1.0.2-beta.54
