注解基础

JDK中的常用注解

元注解

@Documented

表示这个注解应该包含在注解项的文档中

@Inherited

表示当这个注解应用与一个类上时，能够自动被其子类继承

@Retention

表示这个注解在什么时候有效

public enum RetentionPolicy {
    // 源代码级别，只在编译期有效
    SOURCE,

    // 编译后保留在类文件中，但在运行期无效
    CLASS,

    // 运行期有效
    RUNTIME
}

@Target

表示这个注解可以用在哪些地方

public enum ElementType {
    // 类，接口，注解，枚举
    TYPE,

    // 成员字段(包括枚举成员)
    FIELD,

    // 方法
    METHOD,

    // 参数
    PARAMETER,

    // 构造器
    CONSTRUCTOR,

    // 局部变量
    LOCAL_VARIABLE,

    // 注解
    ANNOTATION_TYPE,

    // 包
    PACKAGE,

    TYPE_PARAMETER,

    TYPE_USE,

    // 模块
    MODULE
}
