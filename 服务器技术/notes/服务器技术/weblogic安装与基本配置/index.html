weblogic安装与基本配置

2022-10-11

启动报错：

GC overhead limit exceeded

说明JVM内存溢出，修改 D:\apps\dev\Oracle\Middleware\Oracle_Home\user_projects\domains\base_domain\bin 下的 setDomainEnv.cmd 或 setDomainEnv.sh ，将内存调大

set WLS_MEM_ARGS_64BIT=-Xms512m -Xmx512m

set WLS_MEM_ARGS_32BIT=-Xms512m -Xmx512m

if NOT "%JAVA_VENDOR%"=="HP" (
	if "%VM_TYPE%"=="HotSpot" (
		@REM set WLS_MEM_ARGS_64BIT=-Xms256m -Xmx512m
		@REM set WLS_MEM_ARGS_32BIT=-Xms256m -Xmx512m
		set WLS_MEM_ARGS_64BIT=-Xms1024m -Xmx4096m
		set WLS_MEM_ARGS_32BIT=-Xms1024m -Xmx4096m
	)
)
END
