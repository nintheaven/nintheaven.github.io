配置http反向代理服务器

server { # simple reverse-proxy
    listen       80;
    server_name  domain2.com www.domain2.com;
    access_log   logs/domain2.access.log  main;

    # serve static files
    location ~ ^/(images|javascript|js|css|flash|media|static)/  {
        root    /var/www/virtual/big.server.com/htdocs;
        expires 30d;
    }

    # pass requests for dynamic content to rails/turbogears/zope, et al
    location / {
    	proxy_pass      http://127.0.0.1:8080;
    }
}

nginx报500错误，查看error日志，发现是Permission denied，修改nginx配置中的user为root（默认为nobody）

配置https服务器

安装ssl模块

检查ssl模块是否已安装

nginx -V

如果有 –with-http_ssl_module 选项，则已安装，否则，进行以下操作。

来到nginx源码包路径下

cd /opt/nginx-1.9.5

sudo ./configure --with-http_ssl_module

sudo make

配置静态文件服务器

如果有跨域问题，需要配置额外配置：No ‘Access-Control-Allow-Origin’ header is present on the requested resource

server {
    listen       80;
    server_name  blog-repo.lypzz.com;
    
    location / {
        #root    E:\\blog-repo;
        root    /home/ubuntu/blog/blog-repo;
        # 开启自动生成目录索引功能
        autoindex on;
        # 文件大小：on(默认)时为bytes,off时为KB,MB,GB
        autoindex_exact_size off;
        # 时间显示格式：on时为本地时间格式,off(默认)时为GMT格式
        autoindex_localtime on;
        # 编码
        #charset     gbk;
        charset     utf-8;
        # 设置响应头部，下载文件
        #add_header Content-Disposition attachment;
        
        # 允许跨域的请求
        add_header Access-Control-Allow-Origin *;
        add_header Access-Control-Allow-Methods 'GET, POST, OPTIONS, PUT, PATCH, DELETE';
        add_header Access-Control-Allow-Headers 'DNT,X-Mx-ReqToken,Keep-Alive,User-Agent,X-Requested-With,If-Modified-Since,Cache-Control,Content-Type,Authorization';

        if ($request_method = 'OPTIONS') {
            return 204;
        }
        
        # 禁止访问 403
        location ~* /nginx.conf {
            deny    all;
        }
    }
}

让nginx显示文件内容,而不是下载

nginx默认有如下配置

http {
    include       mime.types;
    default_type  application/octet-stream;
}

其中，mime.types文件中定义了各种文件的MIME类型，如果某种文件类型没有包括在其中，默认使用default_type中配置的方式进行处理，也就是按照二进制字节流处理，即下载文件。

所以，如果想让nginx不自动下载某种文本文件，例如markdown，需要在mime.types文件中增加一条记录

text/plain                            md;

配置负载均衡服务器

upstream big_server_com {
    server 127.0.0.3:8000 weight=5;
    server 127.0.0.3:8001 weight=5;
    server 192.168.0.1:8000;
    server 192.168.0.1:8001;
}

server { # simple load balancing
    listen          80;
    server_name     big.server.com;
    access_log      logs/big.server.access.log main;

    location / {
    	proxy_pass      http://big_server_com;
    }
}
